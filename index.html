Introduction¶

This study analyzes student data from the Polytechnic Institute of Portalegre, Portugal, to identify patterns related to academic failure and support early intervention strategies. The dataset consists of institutional records from undergraduate students enrolled between the 2008/09 and 2018/19 academic years across various disciplines, including agronomy, design, education, nursing, journalism, management, social service, and technologies. By applying exploratory analysis and fairness metrics, the research aims to assess potential biases and ensure equitable predictive insights for student success.

def compute_fairness_metrics(y_true, y_pred):
    """Compute fairness metrics for a given group."""
    accuracy = accuracy_score(y_true, y_pred)
    precision = precision_score(y_true, y_pred, zero_division=0)
    recall = recall_score(y_true, y_pred)  # TPR (Equal Opportunity)
    tn, fp, fn, tp = confusion_matrix(y_true, y_pred).ravel()
    fpr = fp / (fp + tn) if (fp + tn) > 0 else 0  # False Positive Rate
    fnr = fn / (fn + tp) if (fn + tp) > 0 else 0  # False Negative Rate
    treatment_equality = fnr / fpr if fpr > 0 else np.inf  # Treatment Equality

    return {
        "Accuracy": accuracy,
        "Precision": precision,
        "TPR": recall,
        "FPR": fpr,
        "FNR": fnr,
        "Treatment Equality": treatment_equality,
        "Demographic Parity": np.mean(y_pred)
    }

def compute_all_fairness_metrics(y_true_g1, y_pred_g1, y_true_g0, y_pred_g0):
    metrics_g1 = compute_fairness_metrics(y_true_g1, y_pred_g1)
    metrics_g0 = compute_fairness_metrics(y_true_g0, y_pred_g0)

    fairness_metrics = {
        "Demographic Parity": (metrics_g1["Demographic Parity"], metrics_g0["Demographic Parity"]),
        "Accuracy Parity": (metrics_g1["Accuracy"], metrics_g0["Accuracy"]),
        "Predictive Value Parity": (metrics_g1["Precision"], metrics_g0["Precision"]),
        "Equal Opportunity (TPR)": (metrics_g1["TPR"], metrics_g0["TPR"]),
        "Equalized Odds": ((metrics_g1["TPR"], metrics_g1["FPR"]), (metrics_g0["TPR"], metrics_g0["FPR"])),
        "Treatment Equality": (metrics_g1["Treatment Equality"], metrics_g0["Treatment Equality"])
    }

    return fairness_metrics

def compute_accuracy(y_test_g1, y_pred_g1, y_test_g0, y_pred_g0):
  # Calculate accuracy for each group
  accuracy_g1 = accuracy_score(y_test_g1, y_pred_g1)
  accuracy_g0 = accuracy_score(y_test_g0, y_pred_g0)

  # Combine predictions and true labels for overall accuracy
  y_test_combined = np.concatenate([y_test_g1, y_test_g0])
  pre_combined = np.concatenate([y_pred_g1, y_pred_g0])
  overall_accuracy = accuracy_score(y_test_combined, pre_combined)

  print(f"Accuracy for group 1 (Displaced): {accuracy_g1}")
  print(f"Accuracy for group 0 (Not Displaced): {accuracy_g0}")
  print(f"Overall Accuracy: {overall_accuracy}")

def compute_all_fairness_metrics1(y_true_g1, y_pred_g1, y_true_g0, y_pred_g0):
    metrics_g1 = compute_fairness_metrics(y_true_g1, y_pred_g1)
    metrics_g0 = compute_fairness_metrics(y_true_g0, y_pred_g0)

    fairness_metrics = {
        "Demographic Parity": (metrics_g1["Demographic Parity"], metrics_g0["Demographic Parity"], abs(metrics_g1["Demographic Parity"] / metrics_g0["Demographic Parity"])),
        "Accuracy Parity": (metrics_g1["Accuracy"], metrics_g0["Accuracy"], abs(metrics_g1["Accuracy"] / metrics_g0["Accuracy"])),
        "Predictive Value Parity": (metrics_g1["Precision"], metrics_g0["Precision"], abs(metrics_g1["Precision"] / metrics_g0["Precision"])),
        "Equal Opportunity (TPR)": (metrics_g1["TPR"], metrics_g0["TPR"], abs(metrics_g1["TPR"] / metrics_g0["TPR"])),
        "Equalized Odds": ((metrics_g1["TPR"], metrics_g1["FPR"]), (metrics_g0["TPR"], metrics_g0["FPR"]), (abs(metrics_g1["TPR"] / metrics_g0["TPR"]), abs(metrics_g1["FPR"] / metrics_g0["FPR"]))),
        "Treatment Equality": (metrics_g1["Treatment Equality"], metrics_g0["Treatment Equality"], abs(metrics_g1["Treatment Equality"] / metrics_g0["Treatment Equality"]))
    }

    return fairness_metrics

def get_threshold_preds(thresh_g1, thresh_g0, scores, displaced):
    preds = []
    for score, displaced in zip(scores, displaced):
        if displaced == 1:
            preds.append(score >= thresh_g1)
        elif displaced == 0:
            preds.append(score >= thresh_g0)
        else:
            raise ValueError(f"Unexpected Displaced value: {displaced}")
    return preds

def get_rates(displaced_group, labels, preds):
    def calc_rates(group):
        tp = sum(d == group and label and pred for d, label, pred in zip(displaced_group, labels, preds))
        fp = sum(d == group and not label and pred for d, label, pred in zip(displaced_group, labels, preds))
        tn = sum(d == group and not label and not pred for d, label, pred in zip(displaced_group, labels, preds))
        fn = sum(d == group and label and not pred for d, label, pred in zip(displaced_group, labels, preds))

        tpr = tp / (tp + fn) if (tp + fn) else 0
        fpr = fp / (fp + tn) if (fp + tn) else 0
        return tpr, fpr

    tpr_g1, fpr_g1 = calc_rates(1)
    tpr_g0, fpr_g0 = calc_rates(0)

    return tpr_g1, tpr_g0, fpr_g1, fpr_g0

def get_rates_and_threshold_together(X, y, displaced_group):
    scores = mod.predict_proba(X)[:, 1]

    thresh_g1, thresh_g0 = 0.5, 0.5
    step = 0.01

    while thresh_g1 > 0 and thresh_g0 > 0:
        preds = get_threshold_preds(thresh_g1, thresh_g0, scores, displaced_group)
        tpr_g1, tpr_g0, fpr_g1, fpr_g0 = get_rates(displaced_group, y, preds)

        if abs(tpr_g1 - tpr_g0) < 0.01:
            break

        if tpr_g1 < tpr_g0:
            thresh_g1 -= step
            thresh_g0 += step
        elif tpr_g0 < tpr_g1:
            thresh_g0 -= step
            thresh_g1 += step

    final_rates = [tpr_g1, tpr_g0, fpr_g1, fpr_g0]

    return final_rates, thresh_g1, thresh_g0

accuracies_sorted = dp_differences_and_accuracies[:, 1]

# Fit a polynomial curve (e.g., 2nd degree)
degree = 2  # You can change this to 3 for a cubic curve
coefficients = np.polyfit(dp_differences_sorted, accuracies_sorted, degree)
best_fit_curve = np.poly1d(coefficients)

# Generate points for the best-fit curve
x_curve = np.linspace(min(dp_differences), max(dp_differences), 100)
y_curve = best_fit_curve(x_curve)

# Plot the data points and the best-fit curve
plt.figure(figsize=(10, 6))
plt.scatter(dp_differences, accuracies, color='blue', label='Fairness-Aware Models')
